Standalone Examples » Update Data

This example shows how to dynamically change the data set. 

var data = [
  { label: "1", value: 1 },
  { label: "2", value: 4 },
  { label: "3", value: 3 }
];

var pie = new d3pie("pie", {
  data: {
    content: data
  }
});

$(function() {
  var num = 4;
  $("#addData").on("click", function() {
    data.push({
      label: num.toString(),
      value: Math.floor(Math.random() * 10) + 1
    });

    pie.updateProp("data.content", data);
    num++;
  });
});
